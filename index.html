ðŸŒ± BEGINNER LEVEL 18 concepts
CREATE DATABASE
Creates a new database
CREATE DATABASE company;
USE
Switches to a specific database
USE company;
CREATE TABLE
Creates a new table structure
CREATE TABLE employees (id INT, name VARCHAR(50));
INSERT INTO
Adds new records to a table
INSERT INTO employees VALUES (1, 'Alice');
SELECT
Retrieves data from tables
SELECT * FROM employees;
WHERE
Filters records based on conditions
SELECT * FROM employees WHERE age = 25;
ORDER BY
Sorts results in ascending/descending order
SELECT * FROM employees ORDER BY age;
Comparison Operators
Compare values (=, <>, >, <, >=, <=)
SELECT * FROM employees WHERE age > 25;
IN
Matches any value in a list
SELECT * FROM employees WHERE age IN (25, 30);
BETWEEN
Filters values within a range
SELECT * FROM employees WHERE age BETWEEN 25 AND 35;
LIKE
Pattern matching with wildcards
SELECT * FROM employees WHERE name LIKE 'A%';
IS NULL / IS NOT NULL
Checks for NULL values
SELECT * FROM employees WHERE dept IS NOT NULL;
AND / OR
Combines multiple conditions
SELECT * FROM employees WHERE age > 25 AND dept = 'IT';
UPDATE
Modifies existing records
UPDATE employees SET age = 26 WHERE id = 1;
DELETE
Removes records from table
DELETE FROM employees WHERE id = 1;
DROP TABLE
Removes entire table and data
DROP TABLE employees;
TRUNCATE
Removes all records, keeps structure
TRUNCATE TABLE employees;
Modulo (%)
Returns remainder of division
SELECT * FROM employees WHERE id % 2 = 0;
ðŸ”¥ INTERMEDIATE LEVEL 16 concepts
ALTER TABLE ADD
Adds new column to existing table
ALTER TABLE employees ADD phone VARCHAR(15);
ALTER TABLE DROP
Removes column from table
ALTER TABLE employees DROP COLUMN phone;
INNER JOIN
Returns matching records from both tables
SELECT * FROM employees e INNER JOIN projects p ON e.id = p.emp_id;
LEFT JOIN
Returns all from left table, matched from right
SELECT * FROM employees e LEFT JOIN projects p ON e.id = p.emp_id;
RIGHT JOIN
Returns all from right table, matched from left
SELECT * FROM employees e RIGHT JOIN projects p ON e.id = p.emp_id;
FULL OUTER JOIN
Returns all records from both tables
SELECT * FROM employees e FULL OUTER JOIN projects p ON e.id = p.emp_id;
GROUP BY
Groups rows by specified columns
SELECT dept, COUNT(*) FROM employees GROUP BY dept;
COUNT()
Counts number of rows
SELECT COUNT(*) FROM employees;
SUM()
Calculates sum of numeric column
SELECT SUM(salary) FROM employees;
AVG()
Calculates average of numeric column
SELECT AVG(salary) FROM employees;
MAX()
Finds maximum value in column
SELECT MAX(salary) FROM employees;
MIN()
Finds minimum value in column
SELECT MIN(salary) FROM employees;
HAVING
Filters grouped results
SELECT dept, COUNT(*) FROM employees GROUP BY dept HAVING COUNT(*) > 1;
DISTINCT
Returns unique values only
SELECT DISTINCT dept FROM employees;
TOP
Limits number of returned rows
SELECT TOP 2 * FROM employees;
Self JOIN
Joins table with itself
SELECT e1.name, e2.name FROM employees e1 JOIN employees e2 ON e1.dept = e2.dept;
âš¡ ADVANCED LEVEL 14 concepts
NOT NULL
Ensures column cannot have NULL values
CREATE TABLE test (id INT NOT NULL);
UNIQUE
Ensures all values in column are unique
CREATE TABLE test (email VARCHAR(50) UNIQUE);
CHECK
Validates data based on condition
CREATE TABLE test (age INT CHECK(age > 0));
DEFAULT
Sets default value for column
CREATE TABLE test (status VARCHAR(10) DEFAULT 'Active');
PRIMARY KEY
Uniquely identifies each record
CREATE TABLE test (id INT PRIMARY KEY);
FOREIGN KEY
Links to primary key of another table
CREATE TABLE orders (emp_id INT FOREIGN KEY REFERENCES employees(id));
CREATE INDEX
Creates index for faster queries
CREATE INDEX idx_name ON employees(name);
UNION
Combines results, removes duplicates
SELECT name FROM employees UNION SELECT proj_name FROM projects;
UNION ALL
Combines results, keeps duplicates
SELECT name FROM employees UNION ALL SELECT proj_name FROM projects;
CREATE VIEW
Creates virtual table based on query
CREATE VIEW emp_view AS SELECT name, dept FROM employees;
Subquery
Query within another query
SELECT * FROM employees WHERE id IN (SELECT emp_id FROM projects);
CASE WHEN
Conditional logic in SELECT
SELECT CASE WHEN age > 30 THEN 'Senior' ELSE 'Junior' END FROM employees;
EXISTS
Tests for existence of subquery results
SELECT * FROM employees e WHERE EXISTS (SELECT 1 FROM projects p WHERE p.emp_id = e.id);
IDENTITY
Auto-incrementing column
CREATE TABLE test (id INT IDENTITY(1,1));
